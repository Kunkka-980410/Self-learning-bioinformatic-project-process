```python
import matplotlib.pyplot as plt
import numpy as np

# 设置阈值
fc_threshold = 1       # log2FC > 1 或 < -1
p_threshold = 0.05     # p_adj < 0.05

# 添加一列标记基因状态
def label_gene(row):
    if row['p_adj'] < p_threshold:
        if row['log2FC'] > fc_threshold:
            return 'Up'
        elif row['log2FC'] < -fc_threshold:
            return 'Down'
    return 'NotSig'

results['status'] = results.apply(label_gene, axis=1)

# 设置颜色
colors = {'Up': 'red', 'Down': 'blue', 'NotSig': 'gray'}

# 绘图
plt.figure(figsize=(10, 6))
for status in ['Up', 'Down', 'NotSig']:
    subset = results[results['status'] == status]
    plt.scatter(subset['log2FC'], -np.log10(subset['p_value']),
                c=colors[status], label=status, alpha=0.6)

# 添加阈值线
plt.axhline(-np.log10(p_threshold), color='black', linestyle='--', linewidth=1)
plt.axvline(fc_threshold, color='black', linestyle='--', linewidth=1)
plt.axvline(-fc_threshold, color='black', linestyle='--', linewidth=1)

# 标注显著基因（可选）
for i, row in results.iterrows():
    if row['status'] != 'NotSig' and abs(row['log2FC']) > 2:
        plt.text(row['log2FC'], -np.log10(row['p_value']), row['Geneid'],
                 fontsize=8, ha='center', va='bottom')

# 图形美化
plt.xlabel('log2 Fold Change')
plt.ylabel('-log10(p-value)')
plt.title('Volcano Plot: IVDD vs Normal')
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.show()
```
